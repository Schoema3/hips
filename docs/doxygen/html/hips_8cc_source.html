93
94 int maxSc = 1.0;
95

97 maxSc = ScHips[i]>maxSc ? ScHips[i] : maxSc;

117 levelLengths[i] = domainLength * pow(Afac,i);
118 levelTaus[i] = tau0 * pow(levelLengths[i]/domainLength, 2.0/3.0) / C_param;

121 }

131 }
132

134

221std::vector<double> hips::projection(std::vector<double> &vcfd, std::vector<double> &weight) {
222
223 xc = setGridCfd(weight); //populate the physical domain for flow particles
224 xh = setGridHips(nparcels); //populate the physical domain for hips parcels
225
226 int nc = xc.size() - 1;
227 int nh = xh.size() - 1;
228
229 std::vector<double> vh(nh, 0.0);
230 int jprev = 0;
231
232 for(int i = 0; i < nh; i++) {
233 for(int j = jprev + 1; j <= nc; ++j) {
234 if(xc[j] <= xh[i + 1]) {
235 double d1 = xc[j] - xc[j - 1];
236 double d2 = xc[j] - xh[i];
237 double d = std::min(d1, d2);
238
239 vh[i] += vcfd[j - 1] * d;
240 }
241
242 else {
243 double d1 = xh[i + 1] - xc[j - 1];
244 double d2 = xh[i + 1] - xh[i];
245 double d = std::min(d1, d2);
246
247 vh[i] += vcfd[j - 1] * d;
248 jprev = j - 1;
249 break;
250 }
251 }
252
253 vh[i] /= (xh[i + 1] - xh[i]);
254 }
255 return vh;
256}

279std::pair<std::vector<double>, std::vector<double>> hips::projection(std::vector<double> &vcfd, std::vector<double> &weight, const std::vector<double> &density) {
280
281 std::vector<double> xc = setGridCfd(weight);
282 std::vector<double> xh = setGridHips(nparcels);
283
284 // Initialize variables
285 int nc = xc.size() - 1;
286 int nh = xh.size() - 1;
287
288 std::vector<double> vh(nh, 0.0);
289 std::vector<double> rho_h(nh, 0.0);
290 int jprev = 0;
291
292 // Main loop
293 for (int i = 0; i < nh; i++) {
294 double total_dx = 0.0;
295
296 for (int j = jprev + 1; j <= nc; ++j) {
297 if (xc[j] <= xh[i + 1]) {
298 double d = std::min(xc[j] - xc[j - 1], xc[j] - xh[i]);
299 total_dx += d;
300 rho_h[i] += density[j - 1] * d;
301 vh[i] += density[j - 1] * vcfd[j - 1] * d;
302 }
303
304 else {
305 double d = std::min(xh[i + 1] - xc[j - 1], xh[i + 1] - xh[i]);
306 total_dx += d;
307 rho_h[i] += density[j - 1] * d;
308 vh[i] += density[j - 1] * vcfd[j - 1] * d;
309 jprev = j - 1;
310 break;
311 }
312 }
313
314 // Normalize results
315 rho_h[i] /= (xh[i + 1] - xh[i]);
316 vh[i] /= rho_h[i] * (xh[i + 1] - xh[i]);
317 }
318
319 return {vh, rho_h};
320}

382void hips::calculateSolution(const double tRun, bool shouldWriteData) {
383
384 unsigned long long nEddies = 0; // number of eddy events
385 int fileCounter = 0; // number of data files written
386 int iLevel; // tree level of EE with top at iLevel=0
387 int iTree; // one of two subtrees involved in swap at iLevel
388 dtEE;
389 time = 0.0; // initialize simulation time
390
391 sample_hips_eddy(dtEE, iLevel); // get first EE at time 0+dtEE
392 nEddies++;
393
394 while (time+dtEE<=tRun) {
395 time += dtEE;
396 selectAndSwapTwoSubtrees(iLevel, iTree);
397 advanceHips(iLevel, iTree); // reaction and micromixing (if needed) to t=time
398
399 sample_hips_eddy(dtEE, iLevel);
400
401 nEddies++;
402 //cout<<"-------------------------------------"<<nEddies<<endl;
403 if(shouldWriteData && nEddies %20 == 0) writeData(++fileCounter, time);
404 }
405 time = tRun;
406 iLevel = 0; iTree = 0;
407 if(performReaction)
408 reactParcels_LevelTree(iLevel, iTree); // react all parcels up to end time
409}

418void hips::sample_hips_eddy(double &dtEE, int &iLevel) {
419
420 static double c1 = 1.0 - pow(2.0, 5.0/3.0*(iEta+1));
421 static double c2 = pow(2.0, Nm2) - pow(2.0, iEta+1);
422 static double c3 = pow(2.0, iEta+1);
423
424 //--------------- time to next eddy
425
426 double r = rand.getRand();
427 dtEE = -log(r)/eddyRate_total;
428
429 //----------------- get eddy level
430
431 r = rand.getRand();
432
433 if ( r <= eddyRate_inertial/eddyRate_total) { // inertial region
434 r = rand.getRand();
435 iLevel = ceil(3.0/5.0*log2(1.0-r*c1) - 1.0);
436 if (iLevel < 0) iLevel = 0;
437 if (iLevel > iEta) iLevel = iEta;
438 }
439
440 else { // "Batchelor" region
441 r = rand.getRand();
442 iLevel = ceil(log2(r*c2 + c3) - 1.0);
443 if(iLevel < iEta+1) iLevel = iEta+1;
444 if(iLevel > Nm3) iLevel = Nm3;
445 }
446
447 return;
448}

500void hips::selectAndSwapTwoSubtrees(const int iLevel, int &iTree) {
501
502 iTree = rand.getRandInt((1 << iLevel)-1);
503 int zero_q = rand.getRandInt(1); // 0q where q is 0 or 1
504 int one_r = 2 + rand.getRandInt(1); // 1r where r is 0 or 1
505
506 int Qstart = (zero_q << (Nm3-iLevel)) + (iTree << (Nm1-iLevel)); // starting index of Q parcels
507 int Rstart = (one_r << (Nm3-iLevel)) + (iTree << (Nm1-iLevel)); // starting index of R parcels
508 int nPswap = 1 << (Nm3-iLevel); // number of parcels that will be swapped
509
510 int Qend = Qstart + nPswap; // inclusive indices are Qstart to Qend-1
511 int Rend = Rstart + nPswap; // inclusive indices are Rstart to Rend-1
512 vector<int> aa(pLoc.begin()+Qstart, pLoc.begin()+Qend);
513 copy(pLoc.begin()+Rstart, pLoc.begin()+Rend, pLoc.begin()+Qstart); // python: pLoc[Qstart:Qend]=pLoc[Rstart:Rend]
514 copy(aa.begin(), aa.end(), pLoc.begin()+Rstart); // python: pLoc[Rstart:Rend]=aa
515}

527
528 if (forceTurb==2 && iLevel==0) // forcing for statistically stationary

531 bool rxnDone = false; // react all variables once

658void hips::forceProfile() {
659 // Loop through each variable in the HiPS profile
660 for (int k = 0; k < varData.size(); k++) {
661 double s; // Temporary variable for summation
662
663 //---------- Force the left half of parcels to average 0 ----------
664
665 for (int i = 0; i < nparcels >> 1; i++)
666 s += varData[k][0][pLoc[i]]; // Calculate the sum of values in the left half of parcels
667
668 s /= (nparcels >> 1); // Calculate the average of values in the left half of parcels
669
670 for (int i = 0; i < nparcels >> 1; i++)
671 varData[k][0][pLoc[i]] += (-s - 0.0); // Adjust values in the left half of parcels to achieve an average of 0
672
673 //---------- Force the right half of parcels to average 1 ----------
674 s = 0.0;
675
676 for (int i = nparcels >> 1; i < nparcels; i++)
677 s += varData[k][0][pLoc[i]]; // Calculate the sum of values in the right half of parcels
678
679 s /= (nparcels >> 1); // Calculate the average of values in the right half of parcels
680
681 for (int i = nparcels >> 1; i < nparcels; i++)
682 varData[k][0][pLoc[i]] += (-s + 1.0); // Adjust values in the right half of parcels to achieve an average of 1
683 }
684}
